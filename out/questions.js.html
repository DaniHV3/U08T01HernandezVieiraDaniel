/**
 * Obtenemos los paises aleatorios, 5 en este caso
 */
export function getCountries(data) {
  var countries = data.map((e) => e.name); //Nombres únicos
  var countriesRandom = [];
  while (countries.length != 2) {
    var numRandom = getRandomArbitrary(0, countries.length);
    countriesRandom.push(countries[numRandom]);
    countries.splice(numRandom, 1);
  }
  return countriesRandom;
}

/**
 * Obtenemos las ciudades aleatorias de los paises obtenidos
 * @param {*} randomCountries paises obtenidas
 */
export function getCities(data, randomCountries) {
  var contCountry;
  var randomCity = [];
  randomCountries.forEach(function (country) {
    contCountry = data.filter((e) => e.name == country);
    var randomCitie =
      contCountry[0].cities[getRandomArbitrary(0, contCountry[0].cities.length)]
        .name;

    randomCity.push(randomCitie);
  });
  return randomCity;
}

/**
 * Buscamos el un objeto a partir de un país
 * @param {*} nameCountry País
 */
export function objectFinder(data, nameCountry) {
  var contCountry = data.filter((e) => e.name == nameCountry);
  return contCountry;
}
function getRandomArbitrary(min, max) {
  return Math.floor(Math.random() * (max - min)) + min;
}
